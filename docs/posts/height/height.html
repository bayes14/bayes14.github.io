with pm.Model() as basic_model:
    sigma = pm.Exponential("sigma", 10)
    mu = pm.Normal('mu', mu=30, sigma=1)
    obs = pm.Normal("mass", mu=mu, sigma=sigma, observed=data.weight)
    
    prior = pm.sample_prior_predictive()
    idata_basic = pm.sample()
